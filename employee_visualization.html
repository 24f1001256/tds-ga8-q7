const data = [5, 7, 9, 10, 12, 12, 13, 15, 18, 20, 21, 22, 23, 24, 30, 32]; function createHistogram(data) { const ctx = document.getElementById("histChart").getContext("2d"); // Define 5 bins const binCount = 5; const min = Math.min(...data); const max = Math.max(...data); const binSize = (max - min) / binCount; const bins = Array(binCount).fill(0); data.forEach(v => { const index = Math.min( Math.floor((v - min) / binSize), binCount - 1 ); bins[index]++; }); const labels = bins.map((_, i) => (min + i * binSize).toFixed(1) + " - " + (min + (i + 1) * binSize).toFixed(1) ); new Chart(ctx, { type: "bar", data: { labels: labels, datasets: [{ label: "Frequency", data: bins, borderWidth: 1 }] }, options: { scales: { y: { beginAtZero: true } } } }); } createHistogram(data);
