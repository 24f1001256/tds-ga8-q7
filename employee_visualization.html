import pandas as pd
import matplotlib.pyplot as plt

# Sample employee dataset
data = {
    "employee_id": ["EMP001","EMP002","EMP003","EMP004","EMP005"],
    "department": ["IT","Sales","Operations","Marketing","Finance"],
    "region": ["Africa","Latin America","Africa","North America","North America"],
    "performance_score": [85.94,77.79,60.81,60.97,87.44],
    "years_experience": [14,14,11,2,10],
    "satisfaction_rating": [3.1,3.4,3.9,3.7,3.4]
}

df = pd.DataFrame(data)

# Frequency count for Finance department
finance_count = df[df["department"] == "Finance"].shape[0]
print("Finance Department Count:", finance_count)

# Histogram of departments
plt.figure(figsize=(8, 6))
df["department"].value_counts().plot(kind="bar")
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Frequency")
plt.savefig("department_histogram.png")
